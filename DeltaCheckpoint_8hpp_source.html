222 void traceValue(std::ostream& o, const std::vector<ArchData*>& dats,
223 const ArchData* container, uint32_t offset, uint32_t size)
224 {
225 std::stack<DeltaCheckpoint*> dcps = getHistoryChain();
226
227 std::vector<std::pair<uint8_t,bool>> bytes; // pair<value,valid>
228 bytes.resize(size, decltype(bytes)::value_type(0,false));
229
230 constexpr uint32_t BUF_SIZE = 8192*2;
231 std::unique_ptr<char[]> buf(new char[BUF_SIZE]); // Line-reading buffer
232
233 while(!dcps.empty()){
234 DeltaCheckpoint* d = dcps.top();
235 o << "trace: Checkpoint " << d->getDeletedRepr() << (d->isSnapshot()?" (snapshot)":"") << std::endl;
236 dcps.pop();
237 d->data_.prepareForLoad();
238 bool found_ad = false;
239 bool changed = false;
240 for(ArchData* ad : dats){
241 if(ad == container){
242 found_ad = true;
243 if(d->isSnapshot()){
244 for(auto &x : bytes){
245 x.second = false; // Invalidate result due to snapshot load
246 }
247 }
248 }
249 while(1){
250 auto ln_idx = d->data_.getNextRestoreLine();
251 if(ln_idx == ArchData::INVALID_LINE_IDX){
252 if(ad == container){
253 //if(changed == false){
254 // o << "trace: No data for this value" << std::endl;
255 //}
256 }
257 break; // Done with this ArchData
258 }
259 auto ln_off = ln_idx * ad->getLineSize();
260 sparta_assert(BUF_SIZE > ad->getLineSize(),
261 "Cannot trace value on ArchDatas with line sizes > " << BUF_SIZE << " (" << ad->getLineSize() << ")");
262 d->data_.copyLineBytes((char*)buf.get(), ad->getLineSize()); // Need to read regardless of usefulness of data
263 if(ad == container){
264 //o << "trace: Contains data for line idx " << std::dec << ln_idx
265 // << " offsets [" << ln_off << "," << ln_off+ad->getLineSize() << ")" << std::endl;
266 if(offset >= ln_off && offset < ln_off + ad->getLineSize()){
267 sparta_assert(offset+size < ln_off + ad->getLineSize(),
268 "Cannot trace value which spans multiple lines!");
269 sparta_assert(changed == false,
270 "Value being traced changed twice in the same checkpoint");
271 changed = true;
272 auto off_in_line = offset - ln_off;
273 o << "trace: Value changed (line " << std::dec << ln_idx << ")" << std::endl;
274 for(uint32_t i=0; i<bytes.size(); i++){
275 bytes[i].first = buf.get()[i+off_in_line];
276 bytes[i].second = true; // Valid
277 }
278 }
279 }
280 }
281 }
282 if(!found_ad){
283 o << "trace: Could not find selected ArchData " << (const void*)container << " in this checkpoint!" << std::endl;
284 }
285 o << "trace: Value:";
286 for(uint32_t i=0; i<bytes.size(); i++){
287 if(bytes[i].second){
288 o << ' ' << std::setfill('0') << std::setw(2) << std::hex << (uint16_t)bytes[i].first;
289 }else{
290 o << " xx"; // Invalid
291 }
292 }
293 o << std::endl;
294 }
295 o << std::endl;
296 }

408 bool canDelete() const noexcept {
409 if(!isFlaggedDeleted()){
410 return false;
411 }
412 for(auto d : getNexts()){
413 const DeltaCheckpoint* dcp = static_cast<const DeltaCheckpoint*>(d);
414 if(!dcp->canDelete() && !dcp->isSnapshot()){
415 return false;
416 }
417 }
418 return true; // This checkpoint was flagged deleted

543 void loadState(const std::vector<ArchData*>& dats) {
544 data_.prepareForLoad();
545 sparta_assert(data_.good(),
546 "Attempted to loadState from a DeltaCheckpoint with a bad data buffer");
547 if(isSnapshot()){
548 for(ArchData* ad : dats){
549 //std::cout << "Restoring for ArchData: " << (void*)ad << " " << ad->getOwnerNode()->getLocation() << std::endl;
550 ad->restoreAll(data_);
551 }
552 }else{
553 for(ArchData* ad : dats){
554 //std::cout << "Restoring for ArchData: " << (void*)ad << " " << ad->getOwnerNode()->getLocation() << std::endl;
555 ad->restore(data_);
556 }
557 }
