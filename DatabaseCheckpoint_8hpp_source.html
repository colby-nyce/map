235 template <typename Archive>
236 inline void ChkptWindow::serialize(Archive& ar, const unsigned int /*version*/) {
237 ar & start_chkpt_id;
238 ar & end_chkpt_id;
239 ar & start_tick;
240 ar & end_tick;
241
242 if (chkpts.empty()) {
243 // We are loading a checkpoint window from disk
244 const auto num_chkpts = end_chkpt_id - start_chkpt_id + 1;
245 for (size_t i = 0; i < num_chkpts; ++i) {
246 chkpts.emplace_back(new DatabaseCheckpoint);
247 ar & *chkpts.back();
248 }
249 } else {
250 // We are saving a checkpoint window to disk
251 for (auto& chkpt : chkpts) {
252 ar & *chkpt;
253 }
254 }
255 }
