237 template <typename Archive>
238 inline void ChkptWindow::serialize(Archive& ar, const unsigned int /*version*/) {
239 ar & start_chkpt_id;
240 ar & end_chkpt_id;
241 ar & start_tick;
242 ar & end_tick;
243
244 if (chkpts.empty()) {
245 // We are loading a checkpoint window from disk
246 const auto num_chkpts = end_chkpt_id - start_chkpt_id + 1;
247 for (size_t i = 0; i < num_chkpts; ++i) {
248 chkpts.emplace_back(new DatabaseCheckpoint);
249 ar & *chkpts.back();
250 }
251 } else {
252 // We are saving a checkpoint window to disk
253 for (auto& chkpt : chkpts) {
254 ar & *chkpt;
255 }
256 }
257 }
