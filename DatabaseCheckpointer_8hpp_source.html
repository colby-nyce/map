554 void setScheduler(sparta::Scheduler& sched)

558
559 AppT* createApp(DatabaseManager* db_mgr, size_t instance_num = 0) override
560 {
561 auto it = roots_by_inst_num_.find(instance_num);
562 if (it == roots_by_inst_num_.end()) {
563 throw sparta::SpartaException(
564 "No TreeNode (ArchData root) set for DatabaseCheckpointer instance number ")
565 << instance_num << ". Did you forget to call setArchDataRoot()?";
566 }
567
568 auto root = it->second;
569
570 // Make the ctor call that the default AppFactory cannot make.
571 return new AppT(db_mgr, *root, scheduler_);
572 }
573
574 void defineSchema(Schema& schema) const override
575 {
576 AppT::defineSchema(schema);
577 }
578
579private:
