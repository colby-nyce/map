195 void deleteCheckpoint(chkpt_id_t id) override {
196
197 // Flag checkpoint as deleted
198 checkpoint_type* d = findCheckpoint_(id);
199 if(!d){
200 throw CheckpointError("Could not delete checkpoint ID=")
201 << id << " because no checkpoint by this ID was found";
202 }
203
204 // Allow deletion and change ID to UNIDENTIFIED_CHECKPOINT.
205 // This is still part of a chain though until there are no
206 // dependencies on it.
207 if(!d->isFlaggedDeleted()){
208 num_dead_checkpoints_++;
209 if(d->isSnapshot()){
210 num_alive_snapshots_--;
211 }
212 num_alive_checkpoints_--;
213 d->flagDeleted();
214 }
215
216 // Delete this and all contiguous previous checkpoint which were
217 // flagged deleted if possible. Stop if current_ is encountered
218 cleanupChain_(d);
219 }

234 void loadCheckpoint(chkpt_id_t id) override {
235 checkpoint_type* d = findCheckpoint_(id);
236 if(!d){
237 throw CheckpointError("Could not load checkpoint ID=")
238 << id << " because no checkpoint by this ID was found";
239 }
240
241 d->load(getArchDatas());
242
243 // Move current to another checkpoint. Anything between head and the
244 // old current_ is fair game for removal if allowed
245 checkpoint_type* rmv = static_cast<checkpoint_type*>(getCurrent_());
246 setCurrent_(d);
247
248 // Restore scheduler tick number
249 if(sched_){
250 sched_->restartAt(getCurrentTick());
251 }
252
253 // Remove all checkpoints which can be. Stop if the new current_ is
254 // encountered again.
255 // Note that is is OK if current_ was moved to a later position in
256 // the chain. No important checkpoints will be removed. The
257 // important thing is never to remove current_.
258 cleanupChain_(rmv);
259 }

522 void cleanupChain_(checkpoint_type* d) {
523
524 // In order to truly delete any checkpoints, we must traverse back
525 // to the previous snapshot (or the head) and forward to the another
526 // snapshot or the end of the chain.
527 // ONLY if both of those points can be reached without encountering
528 // a living checkpoint or the current checkpoint (forward
529 // only) can the whole chain (including the leading shapshot) be
530 // deleted.
531
533
534 if(d == getHead()){
535 // Cannot delete head of checkpoint tree
536 return;
537 }
538
539 // Walk forward to another snapshot or current
540 const bool needed_later = (getCurrent_() == d) || recursForwardFindAlive_(d);
541 if(needed_later) {
542 // Cannot delete because a later living checkpoint (or current) depends on this
543 if(d->isSnapshot()){
544 // This snapshot is needed later. Move to previous delta and work from there
545 d = static_cast<checkpoint_type*>(d->getPrev());
546 }else{
547 return; // This delta is needed. Therefore all preceeding deltas are needed
548 }
549 }
550
551 // Delete backward until current, head, or a non-flagged-deleted checkpoint is hit.
552 // It is possible to fracture the checkpoint tree by deleting a segment
553 // between two snapshots, so prev can end up with nothing leading up to it
554 while(d && d != getHead() && d->isFlaggedDeleted()){
555
556 // If the checkpoint to delete is the current checkpoint, then
557 // We cannot just set current to the previous checkpoint because
558 // we may have run forward and storing a checkpoint in the
559 // future would depend on the checkpoint we are about to delete.
560 // This could be fixed by requiring the next checkpoint to be a
561 // spapshot. Instead, point to the flagged-deleted checkpoint
562 // and do not delete
563 if(getCurrent_() == d){
564 return;
565 }
566
567 checkpoint_type* prev = static_cast<checkpoint_type*>(d->getPrev());
568
569 // If nothing later in the chain (tree) depends on d's data, it can be deleted.
570 // This also patches the checkpoint tree around the deleted checkpoint
572 if(d->canDelete()) {
573 // Get checkpoint id regardless of whether alive or dead
574 chkpt_id_t id = d->getID();
575 if (d->isFlaggedDeleted()) {
576 id = d->getDeletedID();
577 }
578
579 num_dead_checkpoints_--;
580
581 // Erase element in the map
582 auto itr = chkpts_.find(id);
583 sparta_assert(itr != chkpts_.end());
584 chkpts_.erase(itr);
585 }
586
587 d = prev; // Continue until head is reached
588 }
589 }

600 bool recursForwardFindAlive_(checkpoint_type* d) const
601 {
602 const std::vector<Checkpoint*> & nexts = d->getNexts();
603 for(const auto & chkpt : nexts)
604 {
605 checkpoint_type* dc = static_cast<checkpoint_type*>(chkpt);
606 // Only check descendants for snapshot-ness
607 if(dc->isSnapshot()){
608 // Found a live snapshot that ends this branch. d is not needed
609 // after this
610 return false;
611 }
612 if(dc == getCurrent_()){
613 // Found current in this search chain
614 return true;
615 }
616 if(dc->isFlaggedDeleted() == false){
617 // Encountered a checkpoint later in the chain that still
618 // depends on this.
619 return true;
620 }
621
622 // Continue the search recursively
623 if(recursForwardFindAlive_(dc)){
624 return true;
625 }
626 }
627
628 // Found nothing alive.
629 return false;
630 }

682 void createHead_() override {
683 tick_t tick = 0;
684 if(sched_){
685 tick = sched_->getCurrentTick();
686 }
687
688 if(getHead()){
689 throw CheckpointError("Cannot create head at ")
690 << tick << " because a head already exists in this checkpointer";
691 }
692 if(getRoot().isFinalized() == false){
693 CheckpointError exc("Cannot create a checkpoint until the tree is finalized. Attempting to checkpoint from node ");
694 exc << getRoot().getLocation() << " at tick ";
695 if(sched_){
696 exc << tick;
697 }else{
698 exc << "<no scheduler>";
699 }
700 throw exc;
701 }
702
703 checkpoint_type* dcp = new checkpoint_type(getRoot(), getArchDatas(), next_chkpt_id_++, tick, nullptr, true);
704 chkpts_[dcp->getID()].reset(dcp);
705 setHead_(dcp);
706 num_alive_checkpoints_++;
707 num_alive_snapshots_++;
708 setCurrent_(dcp);
709 }

711 chkpt_id_t createCheckpoint_(bool force_snapshot=false) override {
712 bool is_snapshot;
713 checkpoint_type* prev;
714
715 if(next_chkpt_id_ == checkpoint_type::UNIDENTIFIED_CHECKPOINT){
716 throw CheckpointError("Exhausted all ")
717 << checkpoint_type::UNIDENTIFIED_CHECKPOINT << " possible checkpoint IDs. "
718 << "This is likely a gross misuse of checkpointing";
719 }
720
721 // Caller guarantees a head
722 sparta_assert(getHead() != nullptr);
723
724 tick_t tick;
725 if(sched_){
726 tick = sched_->getCurrentTick();
727 }else{
728 tick = 0;
729 }
730
731 if(sched_ && (tick < getHead()->getTick())){
732 throw CheckpointError("Cannot create a new checkpoint at tick ")
733 << tick << " because this tick number is smaller than the tick number of the head checkpoint at: "
734 << getHead()->getTick() << ". The head checkpoint cannot be reset once created, so it should be done "
735 << "at the start of simulation before running. The simulator front-end should do this so this must "
736 << "likely be fixed in the simulator.";
737 }
738
739 if(nullptr == getCurrent_()){
740 // Creating a delta from the head
741 prev = static_cast<checkpoint_type*>(getHead_());
742 is_snapshot = false;
743 }else{
744 if(sched_ && (tick < getCurrent_()->getTick())){
745 throw CheckpointError("Current tick number from sparta scheduler (")
746 << tick << " ) is less than the current checkpoint's tick number ("
747 << getCurrent_()->getTick() << " To create a checkpoint with an earlier tick number, an "
748 << "older checkpoint having a tick number <= the tick number specified here must first be "
749 << "loaded";
750 }
751
752 // Find latest checkpoint <= tick
753
754 prev = static_cast<checkpoint_type*>(getCurrent_());
755 is_snapshot = prev->getDistanceToPrevSnapshot() >= getSnapshotThreshold();
756 }
757
758 checkpoint_type* dcp = new checkpoint_type(getRoot(),
759 getArchDatas(), // Created during createHead
760 next_chkpt_id_++,
761 tick,
762 prev,
763 force_snapshot || is_snapshot);
764 chkpts_[dcp->getID()].reset(dcp);
765 num_alive_checkpoints_++;
766 num_alive_snapshots_ += (dcp->isSnapshot() == true);
767 setCurrent_(dcp);
768
769 if (dcp->isSnapshot()){
770 // Clean up starting with this snapshot and moving back.
771 // May have an opportunity to free older deltas right now
772 // (instead of upon next deletion)
773 cleanupChain_(dcp);
774 }
775
776 return dcp->getID();
777 }
