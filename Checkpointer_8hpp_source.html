178 CheckpointError exc("Cannot create head at ");

523 void dumpBranch(std::ostream& o,
524 const chkpt_id_t chkpt,
525 uint32_t indent,
526 uint32_t pos,
527 std::deque<uint32_t>& continues) {
529 static const std::string SEP_STR = "-> "; // Normal checkpoint chain
530 static const std::string CONT_SEP_STR = "`> "; // Checkpoint branch from higher line
531 assert(SEP_STR.size() == CONT_SEP_STR.size()); // Must be the same length so the layout looks OK
532
533 // Walk through indent and draw continuations
534 uint32_t i = pos;
535 std::remove_reference<decltype(continues)>::type::const_iterator next_cont = continues.begin();
536 for(; i < indent; ++i){
537 if(next_cont != continues.end() && i == *next_cont){
538 o << '|';
539 while(next_cont != continues.end() && *next_cont == i){ // Skip duplicates and move on to next
540 ++next_cont;
541 }
542 }else{
543 o << ' ';
544 }
545 }
546
547 auto nexts = getNextIDs(chkpt);
548 std::stringstream ss;
549
550 // Draw separator between prev checkpoint and this
551 if(next_cont != continues.end() && *next_cont == indent && indent != pos){
552 ss << CONT_SEP_STR;
553 }else{
554 ss << SEP_STR;
555 }
556
557 // Draw box around object if it is current
558 if(current_ && current_->getID() == chkpt){
559 ss << "[ ";
560 }
561
562 dumpCheckpointNode_(chkpt, ss);
563 ss << ' ';
564
565 if(current_ && current_->getID() == chkpt){
566 ss << ']';
567 }
568
569 o << ss.str();
570 i += ss.str().size();
571
572 // Draw all next checkpoints recursively
573 if(nexts.size() > 0){
574 if(nexts.size() > 1){
575 continues.push_back(i); // Push continues first
576 }
577 dumpBranch(o, nexts.front(), i, i, continues); // No indent, no newline
578 decltype(nexts)::const_iterator itr = nexts.begin() + 1;
579 if(itr == nexts.end()){
580 //continues.pop_back();
581 }else{
582 while(itr < nexts.end()){
583 if(itr+1 == nexts.end()){
584 continues.pop_back(); // Do not show this continuation here
585 }
586 o << '\n';
587 dumpBranch(o, *itr, i, 0, continues); // Indent
588 ++itr;
589 }
590 }
591 }

694 void enumerateArchDatas_() {
695 sparta_assert(adatas_.size() == 0, "FastCheckpointer already has a vector of ArchDatas. Cannot re-enumerate");
696
697 adatas_.clear(); // Clear in case invoked again
698
699 // This is a helper for building the adatas_ vector while detecting
700 // dupicate ArchDatas.
701 // Maps an ArchData key to the TreeNode which pointed to it.
702 std::map<ArchData*,TreeNode*> adatas_helper;
703
704 // Recursively walk the tree and add all ArchDatas to adatas_
705 recursAddArchData_(&root_, adatas_helper);
706 }
707
715 void recursAddArchData_(TreeNode* n, std::map<ArchData*,TreeNode*>& adatas_helper) {
716 assert(n);
717 auto adatas = n->getAssociatedArchDatas();
718 for(ArchData* ad : adatas){
719 if(ad != nullptr){
720 auto itr = adatas_helper.find(ad);
721 if(itr != adatas_helper.end()){
722 throw CheckpointError("Found a second reference to ArchData ")
723 << ad << " in the tree: " << root_.stringize() << " . First reference found throgh "
724 << itr->second->getLocation() << " and second found through " << n->getLocation()
725 << " . An ArchData should be findable throug exactly 1 TreeNode";
726 }
727 adatas_.push_back(ad);
728 adatas_helper[ad] = n; // Add to helper map
729 }
730 }
731 for(TreeNode* child : TreeNodePrivateAttorney::getAllChildren(n)){
732 recursAddArchData_(child, adatas_helper);
733 }
734 }
